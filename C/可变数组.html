可变数组

C语言数组在初始化时都会有一个确定的长度

int* array;
// 长度为100
array = new int[100];

但是很多时候大小不好确定，如果分配的过小，很快就会溢出，如果分配的过大，又会导致空间浪费。 很多框架都会提供可变数组，大致原理如下。

维持一个不可变数组，如果这个数组满了，就进行扩容，创建一个新的更大容量的数组来替换原数组，如果现有的数据比现在容量下很多，就创建一个容量较小的数据来替换原数组。

#include <iostream>

class MutableArray {

    // 数组总长度
    int length = 0;
    // 数组已经使用的长度
    int use = 0;
    // 不可变数组指针
    int* mArray;

public:

    MutableArray(int cap) {
        if (cap < 4) {
            cap = 4;
        }

        length = cap;
        mArray = new int[cap];
    };


    void grow() {
        int l = length + (length>>1);
        int* growArray = new int[l];

        for (int i=0; i<use; i++) {
            growArray[i] = mArray[i];
        }

        delete [] mArray;

        length = l;
        mArray = growArray;
    }

    void add(int i) {

        if (use+1 < length) {
            // 直接添加
            mArray[use] = i;
            use++;
        } else if (use+1 == length) {
            // 空间已被用完
            grow();
            add(i);
        }
    }

    void remove(int index) {
        if (index == use -1) {
            // 无需移动
            mArray[index] = 0;
            use--;
        } else {
            for (int i=index; i<use; i++) {
                // 往前移动
                mArray[i] = mArray[i+1];
            }
            use--;
        }

        if (use <= length/4) {
            // 保留1/2的空间，这样可以继续添加元素
            narrow();
        }
    }

    void narrow() {
        int l = length/2;
        int *narrowArray = new int[l];
        for (int i=0; i<use; i++) {
            narrowArray[i] = mArray[i];
        }

        delete [] mArray;
        length = l;
        mArray = narrowArray;
    }

    void log() {
        for (int i=0; i<use; i++) {
            printf("%d,",mArray[i]);
        }
        printf("use=%d length=%d \n",use,length);
    }
};


int main(int argc, const char * argv[]) {
    // insert code here...
    std::cout << "Hello, World!\n";

    MutableArray* mArray = new MutableArray(1);

    mArray->add(1);
    mArray->add(2);
    mArray->add(3);
    mArray->add(4);
    mArray->add(5);
    mArray->add(6);
    mArray->add(7);

    mArray->log();

    mArray->remove(4);
    mArray->remove(3);
    mArray->remove(2);
    mArray->remove(1);
    mArray->remove(0);

    mArray->log();

    return 0;
}
